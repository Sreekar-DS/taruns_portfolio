function filterItems(className, filterId) {
  var filter = document.getElementById(filterId).value;
  var items = document.getElementsByClassName(className);
  for (var i = 0; i < items.length; i++) {
    if (filter === "all") {
      items[i].style.display = "block";
    } else if (items[i].classList.contains(filter)) {
      items[i].style.display = "block";
    } else {
      items[i].style.display = "none";
    }
  }
}

function filterProjects() {
  filterItems('project', 'project-filter');
}

function filterCertifications() {
  filterItems('certification', 'certification-filter');
}

function filterProfessionalDevelopment() {
  filterItems('professional-development', 'professional-development-filter');
}
